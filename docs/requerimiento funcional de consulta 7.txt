Requerimiento funcional de consulta 7

SELECT US.IDENTIFICACION,US.NOMBRE,US.TIPOIDENTIFICACION,US.CORREO,us.idtipo,SUM(RES.COSTO)
FROM USUARIO us 
INNER JOIN RESERVA RES ON RES.ID_USUARIO=US.IDENTIFICACION 
GROUP BY US.IDENTIFICACION,US.NOMBRE,US.TIPOIDENTIFICACION,US.CORREO, us.idtipo 
HAVING SUM(RES.COSTO)>15000000;
