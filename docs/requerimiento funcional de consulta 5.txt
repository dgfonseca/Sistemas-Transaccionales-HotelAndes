SELECT US.IDENTIFICACION, SUM(RES.COSTO)
FROM USUARIO US
INNER JOIN RESERVA RES
ON US.IDENTIFICACION=RES.ID_USUARIO
WHERE (US.IDENTIFICACION= "PARAM") AND (RES.FECHAINICIO BETWEEN "PARAM" AND "PARAM") AND (RES.FECHAFIN BETWEEN "PARAM" AND "PARAM")
GROUP BY US.IDENTIFICACION;
