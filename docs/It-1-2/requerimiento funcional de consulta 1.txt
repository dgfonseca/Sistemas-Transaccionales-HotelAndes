///REQUERIMIENTO FUNCIONAL DE CONSULTA 1///

SELECT HAB.NUMEROHABITACION AS NUMERO_HABITACION, SUM(SERV.COSTO) AS DINERO_RECOLECTADO
FROM HABITACION HAB 
INNER JOIN SIRVEN SIRV ON HAB.NUMEROHABITACION=SIRV.NUMEROHABITACION
INNER JOIN SERVICIOS SERV ON SIRV.IDSERVICIO=SERV.ID
INNER JOIN APARTAN AP ON HAB.NUMEROHABITACION=AP.NUMEROHABITACION
INNER JOIN RESERVA RES ON AP.IDRESERVA=RES.ID
WHERE RES.FECHAINICIO IS BETWEEN "PARAM" AND "PARAM"
GROUP BY HAB.NUMEROHABITACION;
